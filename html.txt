<div class="card">
    <h3>Definite Integral</h3>
    <label>Function f(x):</label>
    <input type="text" id="func" placeholder="e.g., x^2">
    
    <label>Lower Bound (a):</label>
    <input type="number" id="lower" value="0">
    
    <label>Upper Bound (b):</label>
    <input type="number" id="upper" value="1">
    
    <button onclick="calculate()">Calculate Integral</button>
    
    <div id="result">Result: </div>
</div>


<script>
    function calculate() {
        const f_str = document.getElementById('func').value;
        const a = parseFloat(document.getElementById('lower').value);
        const b = parseFloat(document.getElementById('upper').value);


        try {
            // Using numerical integration (simpson's rule approximation)
            // math.js doesn't have a symbolic integral function, 
            // so we evaluate the definite integral numerically.
            const f = math.compile(f_str);
            const steps = 1000;
            const h = (b - a) / steps;
            let sum = 0;


            for (let i = 0; i < steps; i++) {
                let x = a + i * h;
                sum += f.evaluate({x: x}) * h;
            }


            document.getElementById('result').innerText = "Result: " + sum.toFixed(6);
        } catch (e) {
            document.getElementById('result').innerText = "Error: Invalid input";
        }
    }
</script>
